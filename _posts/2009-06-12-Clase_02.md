IC.wc <- function(x, n, conf.level = 0.95) {
    p <- x/n
    q <- 1 - p
    alpha <- 1 - conf.level
    z <- qnorm(p = alpha/2, lower.tail = FALSE)
    const1 <- z * sqrt(z^2 - 2 - 1/n + 4 * p * (n * q + 1))
    const2 <- z * sqrt(z^2 + 2 - 1/n + 4 * p * (n * q - 1))
    L <- (2 * n * p + z^2 - 1 - const1)/(2 * (n + z^2))
    U <- (2 * n * p + z^2 + 1 + const2)/(2 * (n + z^2))
    c(L, U)
}
